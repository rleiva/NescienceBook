\begin{sourcecode}
{\scriptsize \begin{verbatim}
SELECT
    page.page_title, categorylinks.cl_to
FROM
    page, categorylinks
WHERE
    page.page_id = categorylinks.cl_from
AND
     categorylinks.cl_type = 'subcat'
INTO OUTFILE '/data/ArsMagna/mysql/SubcatGraph.csv'
FIELDS ENCLOSED BY '"'
TERMINATED BY ';'
ESCAPED BY '"'
LINES TERMINATED BY '\n';
\end{verbatim}}
